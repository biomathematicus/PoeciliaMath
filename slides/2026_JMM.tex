 	  \begin{itemize}
 	  	\item LLMs generate human-like text but produce \textbf{invalidations} (false/misleading statements). Traditional term ``hallucination'' is too narrow
 	  	\item \textbf{Invalidation} encompasses: factual errors, logical contradictions, format violations, ethical breaches
 	  	\item These errors propagate through interconnected communication systems
 	  \end{itemize}

\section{Framework}

%-------------------------------------------------
\frame {
 	  \frametitle{Discursive Networks}
 	  
 	  \textbf{Definition:} A discursive network is a structure $N = (S, P, I, C, A, B, U, G)$ where:
 	  
 	  \begin{itemize}
 	  	\item $S$ = set of statements 
 	  	\item $P$ = persuasion functions
 	  	\item $I$ = invalidation mechanisms
 	  	\item $C$ = communications between actors
 	  	\item $A$ = set of actors (humans, LLMs)
 	  	\item $B$ = beliefs
 	  	\item $U$ = update rules
 	  	\item $G$ = goal functions
 	  \end{itemize}
 	  
 	  \vspace{0.3cm}
 	  
 	  \textbf{Key insight:} (i) The medium (LLM) acts on the message, not just transmits it. (ii) To communicate is to err: \textipa{[spi:k @ b2g]}
}
%-------------------------------------------------
\frame {
 	  \frametitle{Invalidation Floor Theorem}
 	  Given:
 	  \begin{itemize}
		\item  $\mathrm{KL}(Q \parallel P_{\theta})$: Kullback-Leibler divergence from the reference distribution $Q$ to the model distribution $P_{\theta}$.
 	  	\item $Q$ = training distribution
 	  	\item $P_{\theta}$ = trained model distribution
 	  	\item $S$ = set of all invalidations
 	  	\item $q = Q(S)$ = fraction of invalid content in training
 	  \end{itemize}
 	  
 	  \textbf{Lemma:} If $P_{\theta} \ll Q$ and $\mathrm{KL}(Q \parallel P_{\theta}) < \infty$, then
 	  $$P_{\theta}(S) \geq q \exp\left(-\frac{\mathrm{KL}(Q \parallel P_{\theta})}{q}\right)$$
 	  
 	  
 	  \vspace{0.3cm}
 	  
 	  \textbf{Consequence:} Zero invalidation probability is impossible for finite-loss models
}

%-------------------------------------------------
\frame {
      \frametitle{Verification Advantage Theorem}
      Given:
      \begin{itemize}
        \item $(L_g, Y_g)$,  $(L_v, Y_v)$: joint random variables: length and content
        \item $P_g^{(n)}$,  $P_v^{(n)}$: cond. dist. $P(Y_g|L_g=n)$ \& $P(Y_v|L_v=n)$
        \item $\pi_g(n)$, $\pi_v(n)$: length distributions $P(L_g=n)$ \& $P(L_v=n)$
      \end{itemize}
     
      \textbf{Theorem:} The entropy difference decomposes as:
	 
	  \begin{align}
		H(L_g, Y_g) - H(L_v, Y_v) = &  H(L_g) - H(L_v) + \sum_{n} \pi_g(n)H(Y_g|L_g = n) \nonumber \\
		& - \sum_{n} \pi_v(n)H(Y_v|L_v = n) \nonumber	  
	  \end{align}
      
      \vspace{0.2cm}
      
      \textbf{Consequence:} Detection rates $d$ can systematically exceed fabrication rates $\lambda$
}


%-------------------------------------------------
\frame {
      \frametitle{Verification Advantage: Intuition}
      
      \textbf{Core Insight:} It's easier to spot a mistake than to avoid making one.
      
      \vspace{0.3cm}
      
      \textbf{The Decomposition:}
      \begin{itemize}
        \item Total uncertainty = Length uncertainty + Content uncertainty
        \item Applies identically to generation and verification tasks
        \item But the \emph{magnitudes} differ dramatically in practice
      \end{itemize}
      
      \vspace{0.3cm}
      
      \textbf{Empirical Asymmetry:}
      \begin{itemize}
        \item \textbf{Generation}: Probability spread across thousands of valid phrasings
        \item \textbf{Verification}: $>$90\% of probability mass on 2--3 tokens (Yes/No)
      \end{itemize}
      
      \vspace{0.2cm}
      
      $\Rightarrow$ Verification operates in a \emph{low-entropy regime} where reliable outputs are easier to produce.
}

%-------------------------------------------------
\frame {
      \frametitle{Verification Advantage: Implications}
      
      \textbf{Analogy: Writing vs.\ Proofreading}
      \begin{itemize}
        \item \textbf{Writing}: Vast landscape of choices $\rightarrow$ probability spread thin
        \item \textbf{Proofreading}: Binary classifier at each checkpoint $\rightarrow$ concentrated outputs
      \end{itemize}
      
      \vspace{0.3cm}
      
      \textbf{Why Multi-Agent Verification Works:}
      \begin{itemize}
        \item Agens have nonzero `invalidation floor' (errors inevitable)
        \item But when Agent B \emph{critiques} Agent A, it operates in verification mode
        \item Detection rate $d$ can exceed fabrication rate $\lambda$
      \end{itemize}
      
      \vspace{0.3cm}
      
      \textbf{FOO Algorithm Consequence:}
      \[
        \pi_f^{(n)} = \frac{p + \lambda}{p + \lambda + q + (n-1)d}
      \]
      
      Adding verifiers reduces error \emph{hyperbolically} $\rightarrow$ systematic improvement even with imperfect agents.
}


%-------------------------------------------------
\frame {
 	  \frametitle{Verification Advantage}
 	  
 	  \textbf{Information-theoretic insight:} Verification is easier than generation
 	  
 	  \begin{itemize}
 	  	\item \textbf{Generation:} Explores long tail of output distribution
 	  	\item \textbf{Verification:} Operates in high-probability regions. Critique patterns (``One limitation is...'') are frequent in training data
 	  \end{itemize}
 	  
 	  \vspace{0.5cm}
 	  
 	  \textbf{Empirical evidence:}
 	  \begin{itemize}
 	  	\item Detection rates $d$ usually exceed fabrication rates $\lambda$
 	  \end{itemize}
}

%-------------------------------------------------
\frame {
 	  \frametitle{Single Network Model}
 	  
 	  Binary state system with statements $S = \{r, f\}$ (true/false):
 	  
 	  $$\boldsymbol{\pi}(t+1) = T \boldsymbol{\pi}(t), \quad T = \begin{pmatrix}
 	  1-(p+\lambda) & q \\
 	  p+\lambda & 1-q
 	  \end{pmatrix}$$
 	  
 	  where:
 	  \begin{itemize}
 	  	\item $p$ = flip rate: intrinsic truth $\to$ false 
 	  	\item $q$ = internal self-correction rate  
 	  	\item $\lambda$ = fabrication (invalidation) rate
 	  \end{itemize}
 	  
 	  \textbf{Fixed point:}
 	  $$\pi_f^* = \frac{p+\lambda}{p+\lambda+q}, \quad \pi_r^* = \frac{q}{p+\lambda+q}$$
}

%-------------------------------------------------
\frame {
 	  \frametitle{Cross-Network Detection Model}
 	  
 	  Two networks $N_1, N_2$ with cross-detection probability $d$:
 	  
 	  \textbf{Key dynamics:}
 	  \begin{itemize}
 	  	\item Each network generates fabrications: $X_k(t) \sim \text{Poisson}(\lambda_k T_k(t))$
 	  	\item Cross-detection: $Y_{jk}(t) \sim \text{Binomial}(F_k(t), d_{jk})$
 	  	\item Internal flips: $Z_k(t) \sim \text{Binomial}(T_k(t), p_k)$, $W_k(t) \sim \text{Binomial}(F_k(t), q_k)$
 	  \end{itemize}
 	  
 	  \textbf{Equilibrium false share:}
 	  $$\pi_{F,k}^* = \frac{\lambda_k}{d_{jk}}$$
 	  
 	  \textbf{Condition for improvement:} $\frac{\lambda}{d} < \frac{p+\lambda}{p+\lambda+q}$
}

}

\frame {
 	  \frametitle{Flaws-of-Others (FOO) Algorithm}
 	  
 	  \textbf{Protocol:}
 	  \begin{enumerate}
 	  	\item \textbf{Broadcast:} User task sent to all agents
 	  	\item \textbf{Cross-examination:} Each agent critiques peers' outputs
 	  	\item \textbf{Harmonization:} Aggregate critiques into judgment
 	  	\item \textbf{Revision:} Agents revise based on feedback
 	  	\item Repeat until convergence
 	  \end{enumerate}
 	  
 	  \vspace{0.3cm}
 	  
 	  \textbf{Key features:}
 	  \begin{itemize}
 	  	\item Model-agnostic (any LLM backend)
 	  	\item Configurable agent ensemble
 	  	%\item Cryptographic integrity verification
 	  	\item Implements cross-detection hazard $d$
 	  \end{itemize}
}

\frame {
 	  \frametitle{Agent Requirements}
 	  
 	  \textbf{Theorem:} To achieve false share $\leq \varepsilon$, need at least:
 	  
 	  $$n_{\min} = \left\lceil 1 + \frac{(p+\lambda)\left(\frac{1}{\varepsilon}-1\right) - q}{d} \right\rceil$$
 	  
 	  agents with cross-detection rate $d$.
 	  
 	  \vspace{0.5cm}
 	  
 	  \textbf{Example:} For 5\% error rate with parameters
 	  \begin{itemize}
 	  	\item $p = 0.02$, $q = 0.05$, $\lambda = 0.055$, $d = 0.19$
 	  	\item Need $n_{\min} = 9$ mutually detecting agents
 	  \end{itemize}
 	  
 	  \textbf{Scaling:} Linear growth in $1/\varepsilon$ makes high reliability tractable
}

\frame {
 	  \frametitle{Ethical Concerns}
 	  
 	  \textbf{Epithesis:} Claiming authorship without genuine contribution
 	  
 	  \textbf{Energy costs:} $O(N^2 \log N)$ computational complexity
 	  
 	  \textbf{Epistemic diversity:} Risk of homogenization through consensus
 	  
 	  \textbf{Privacy:} Cross-agent visibility increases attack surface
 	  
 	  \vspace{0.5cm}
 	  
 	  \textbf{Mitigation strategies:}
 	  \begin{itemize}
 	  	\item Cryptographic integrity verification
 	  	\item Governance frameworks for fair access
 	  	\item Diversity-preserving consensus mechanisms
 	  	\item Privacy-preserving critique protocols
 	  \end{itemize}
}

\frame {
 	  \frametitle{Cultural Implications}
 	  
 	  \textbf{We have entered a new cultural mode:}
 	  
 	  \begin{itemize}
 	  	\item Writing is now \textbf{co-constructed} with machines
 	  	\item Authorship becomes \textbf{distributed} across agents
 	  	\item Traditional metrics (originality, citations) need updating
 	  	\item The medium itself \textbf{acts on the message}
 	  \end{itemize}
 	  
 	  \vspace{0.5cm}
 	  
 	  \textbf{Future research directions:}
 	  \begin{enumerate}
 	  	\item Cultural assimilation and ethical frameworks
 	  	\item Redesign of quality metrics for co-creative medium
 	  	\item Governance of distributed agency
 	  \end{enumerate}
